159  bool shouldMove
160  {
161  get
162  {
163  if (!disableDragIfFits) return true;
164 
165  if (mPanel == null) mPanel = GetComponent<UIPanel>();
166  Vector4 clip = mPanel.clipRange;
167  Bounds b = bounds;
168 
169  float hx = clip.z * 0.5f;
170  float hy = clip.w * 0.5f;
171 
172  if (!Mathf.Approximately(scale.x, 0f))
173  {
174  if (b.min.x < clip.x - hx) return true;
175  if (b.max.x > clip.x + hx) return true;
176  }
177 
178  if (!Mathf.Approximately(scale.y, 0f))
179  {
180  if (b.min.y < clip.y - hy) return true;
181  if (b.max.y > clip.y + hy) return true;
182  }
183  return false;
184  }

228  public void RestrictWithinBounds (bool instant)
229  {
230  Vector3 constraint = mPanel.CalculateConstrainOffset(bounds.min, bounds.max);
231 
232  if (constraint.magnitude > 0.001f)
233  {
234  if (!instant && dragEffect == DragEffect.MomentumAndSpring)
235  {
236  // Spring back into place
237  SpringPanel.Begin(mPanel.gameObject, mTrans.localPosition + constraint, 13f);
238  }
239  else
240  {
241  // Jump back into place
242  MoveRelative(constraint);
243  mMomentum = Vector3.zero;
244  mScroll = 0f;
245  }
246  }
247  else
248  {
249  // Remove the spring as it's no longer needed
250  DisableSpring();
251  }

268  public void UpdateScrollbars (bool recalculateBounds)
269  {
270  if (mPanel == null) return;
271 
272  if (horizontalScrollBar != null || verticalScrollBar != null)
273  {
274  if (recalculateBounds)
275  {
276  mCalculatedBounds = false;
277  mShouldMove = shouldMove;
278  }
279 
280  if (horizontalScrollBar != null)
281  {
282  Bounds b = bounds;
283  Vector3 boundsSize = b.size;
284 
285  if (boundsSize.x > 0f)
286  {
287  Vector4 clip = mPanel.clipRange;
288  float extents = clip.z * 0.5f;
289  float min = clip.x - extents - b.min.x;
290  float max = b.max.x - extents - clip.x;
291 
292  if (mPanel.clipping == UIDrawCall.Clipping.SoftClip)
293  {
294  min += mPanel.clipSoftness.x;
295  max -= mPanel.clipSoftness.x;
296  }
297 
298  min = Mathf.Clamp01(min / boundsSize.x);
299  max = Mathf.Clamp01(max / boundsSize.x);
300 
301  float sum = min + max;
302  mIgnoreCallbacks = true;
303  horizontalScrollBar.barSize = 1f - sum;
304  horizontalScrollBar.scrollValue = (sum > 0.001f) ? min / sum : 0f;
305  mIgnoreCallbacks = false;
306  }
307  }
308 
309  if (verticalScrollBar != null)
310  {
311  Bounds b = bounds;
312  Vector3 boundsSize = b.size;
313 
314  if (boundsSize.y > 0f)
315  {
316  Vector4 clip = mPanel.clipRange;
317  float extents = clip.w * 0.5f;
318  float min = clip.y - extents - b.min.y;
319  float max = b.max.y - extents - clip.y;
320 
321  if (mPanel.clipping == UIDrawCall.Clipping.SoftClip)
322  {
323  min += mPanel.clipSoftness.y;
324  max -= mPanel.clipSoftness.y;
325  }
326 
327  min = Mathf.Clamp01(min / boundsSize.y);
328  max = Mathf.Clamp01(max / boundsSize.y);
329  float sum = min + max;
330 
331  mIgnoreCallbacks = true;
332  verticalScrollBar.barSize = 1f - sum;
333  verticalScrollBar.scrollValue = (sum > 0.001f) ? 1f - min / sum : 0f;
334  mIgnoreCallbacks = false;
335  }
336  }
337  }
338  else if (recalculateBounds)
339  {
340  mCalculatedBounds = false;
341  }

349  public void SetDragAmount (float x, float y, bool updateScrollbars)
350  {
351  DisableSpring();
352 
353  Bounds b = bounds;
354  if (b.min.x == b.max.x || b.min.y == b.max.x) return;
355  Vector4 cr = mPanel.clipRange;
356 
357  float hx = cr.z * 0.5f;
358  float hy = cr.w * 0.5f;
359  float left = b.min.x + hx;
360  float right = b.max.x - hx;
361  float bottom = b.min.y + hy;
362  float top = b.max.y - hy;
363 
364  if (mPanel.clipping == UIDrawCall.Clipping.SoftClip)
365  {
366  left -= mPanel.clipSoftness.x;
367  right += mPanel.clipSoftness.x;
368  bottom -= mPanel.clipSoftness.y;
369  top += mPanel.clipSoftness.y;
370  }
371 
372  // Calculate the offset based on the scroll value
373  float ox = Mathf.Lerp(left, right, x);
374  float oy = Mathf.Lerp(top, bottom, y);
375 
376  // Update the position
377  if (!updateScrollbars)
378  {
379  Vector3 pos = mTrans.localPosition;
380  if (scale.x != 0f) pos.x += cr.x - ox;
381  if (scale.y != 0f) pos.y += cr.y - oy;
382  mTrans.localPosition = pos;
383  }
384 
385  // Update the clipping offset
386  cr.x = ox;
387  cr.y = oy;
388  mPanel.clipRange = cr;
389 
390  // Update the scrollbars, reflecting this change
391  if (updateScrollbars) UpdateScrollbars(false);
392  }

469  public void Press (bool pressed)
470  {
471  if (enabled && gameObject.active)
472  {
473  mTouches += (pressed ? 1 : -1);
474  mCalculatedBounds = false;
475  mShouldMove = shouldMove;
476  if (!mShouldMove) return;

481  // Remove all momentum on press
482  mMomentum = Vector3.zero;

485  // Disable the spring movement

487 
488  // Remember the hit position
489  mLastPos = UICamera.lastHit.point;
490 
491  // Create the plane to drag along
492  mPlane = new Plane(mTrans.rotation * Vector3.back, mLastPos);
493  }
494  else if (restrictWithinPanel && mPanel.clipping != UIDrawCall.Clipping.None && dragEffect == DragEffect.MomentumAndSpring)
495  {

505  public void Drag (Vector2 delta)
506  {
507  if (enabled && gameObject.active && mShouldMove)
508  {
509  UICamera.currentTouch.clickNotification = UICamera.ClickNotification.BasedOnDelta;
510 
511  Ray ray = UICamera.currentCamera.ScreenPointToRay(UICamera.currentTouch.pos);
512  float dist = 0f;
513 
514  if (mPlane.Raycast(ray, out dist))
515  {
516  Vector3 currentPos = ray.GetPoint(dist);
517  Vector3 offset = currentPos - mLastPos;
518  mLastPos = currentPos;
519 
520  if (offset.x != 0f || offset.y != 0f)
521  {
522  offset = mTrans.InverseTransformDirection(offset);
523  offset.Scale(scale);
524  offset = mTrans.TransformDirection(offset);
525  }
526 
527  // Adjust the momentum
528  mMomentum = Vector3.Lerp(mMomentum, mMomentum + offset * (0.01f * momentumAmount), 0.67f);
529 
530  // Move the panel
531  MoveAbsolute(offset);
532 
533  // We want to constrain the UI to be within bounds
534  if (restrictWithinPanel &&
535  mPanel.clipping != UIDrawCall.Clipping.None &&
536  dragEffect != DragEffect.MomentumAndSpring)
537  {
538  RestrictWithinBounds(false);
539  }
540  }

562  void LateUpdate ()
563  {
564  // If the panel's geometry changed, recalculate the bounds
565  if (mPanel.changedLastFrame) UpdateScrollbars(true);
566 
567  // Inspector functionality
568  if (repositionClipping)
569  {
570  repositionClipping = false;
571  mCalculatedBounds = false;
572  SetDragAmount(relativePositionOnReset.x, relativePositionOnReset.y, true);
573  }
574 
575  if (!Application.isPlaying) return;
576  float delta = UpdateRealTimeDelta();
577 
578  // Fade the scroll bars if needed
579  if (showScrollBars != ShowCondition.Always)
580  {
581  bool vertical = false;
582  bool horizontal = false;
583 
584  if (showScrollBars != ShowCondition.WhenDragging || mTouches > 0)
585  {
586  vertical = shouldMoveVertically;
587  horizontal = shouldMoveHorizontally;
588  }
589 
590  if (verticalScrollBar)
591  {
592  float alpha = verticalScrollBar.alpha;
593  alpha += vertical ? delta * 6f : -delta * 3f;
594  alpha = Mathf.Clamp01(alpha);
595  if (verticalScrollBar.alpha != alpha) verticalScrollBar.alpha = alpha;
596  }
597 
598  if (horizontalScrollBar)
599  {
600  float alpha = horizontalScrollBar.alpha;
601  alpha += horizontal ? delta * 6f : -delta * 3f;
602  alpha = Mathf.Clamp01(alpha);
603  if (horizontalScrollBar.alpha != alpha) horizontalScrollBar.alpha = alpha;
604  }
605  }
606 
607  // Apply momentum
608  if (mShouldMove && !mPressed)
609  {
610  mMomentum += scale * (-mScroll * 0.05f);
611 
612  if (mMomentum.magnitude > 0.0001f)
613  {
614  mScroll = NGUIMath.SpringLerp(mScroll, 0f, 20f, delta);
615 
616  // Move the panel
617  Vector3 offset = NGUIMath.SpringDampen(ref mMomentum, 9f, delta);
618  MoveAbsolute(offset);
619 
620  // Restrict the contents to be within the panel's bounds
621  if (restrictWithinPanel && mPanel.clipping != UIDrawCall.Clipping.None) RestrictWithinBounds(false);
622  return;
623  }
624  else mScroll = 0f;
625  }
626  else mScroll = 0f;
627 
628  // Dampen the momentum
629  NGUIMath.SpringDampen(ref mMomentum, 9f, delta);
630  }
