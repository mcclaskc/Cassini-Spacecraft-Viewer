115  void Update ()
116  {
117  float delta = UpdateRealTimeDelta();
118 
119  // Advance the sampling factor
120  mFactor += amountPerDelta * delta;
121 
122  // Loop style simply resets the play factor after it exceeds 1.
123  if (style == Style.Loop)
124  {
125  if (mFactor > 1f)
126  {
127  mFactor -= Mathf.Floor(mFactor);
128  }
129  }
130  else if (style == Style.PingPong)
131  {
132  // Ping-pong style reverses the direction
133  if (mFactor > 1f)
134  {
135  mFactor = 1f - (mFactor - Mathf.Floor(mFactor));
136  mAmountPerDelta = -mAmountPerDelta;
137  }
138  else if (mFactor < 0f)
139  {
140  mFactor = -mFactor;
141  mFactor -= Mathf.Floor(mFactor);
142  mAmountPerDelta = -mAmountPerDelta;
143  }
144  }
145 
146  // Calculate the sampling value
147  float val = Mathf.Clamp01(mFactor);
148 
149  if (method == Method.EaseIn)
150  {
151  val = 1f - Mathf.Sin(0.5f * Mathf.PI * (1f - val));
152  if (steeperCurves) val *= val;
153  }
154  else if (method == Method.EaseOut)
155  {
156  val = Mathf.Sin(0.5f * Mathf.PI * val);
157 
158  if (steeperCurves)
159  {
160  val = 1f - val;
161  val = 1f - val * val;
162  }
163  }
164  else if (method == Method.EaseInOut)
165  {
166  const float pi2 = Mathf.PI * 2f;
167  val = val - Mathf.Sin(val * pi2) / pi2;
168 
169  if (steeperCurves)
170  {
171  val = val * 2f - 1f;
172  float sign = Mathf.Sign(val);
173  val = 1f - Mathf.Abs(val);
174  val = 1f - val * val;
175  val = sign * val * 0.5f + 0.5f;
176  }
177  }
178 
179  // Call the virtual update
180  OnUpdate(val);
181 
182  // If the factor goes out of range and this is a one-time tweening operation, disable the script
183  if (style == Style.Once && (mFactor > 1f || mFactor < 0f))
184  {
185  mFactor = Mathf.Clamp01(mFactor);
186 
187  if (string.IsNullOrEmpty(callWhenFinished))
188  {
189  enabled = false;
190  }
191  else
192  {
193  if (eventReceiver != null && !string.IsNullOrEmpty(callWhenFinished))
194  {
195  // Notify the event listener target
196  eventReceiver.SendMessage(callWhenFinished, this, SendMessageOptions.DontRequireReceiver);
197  }
198 
199  // Disable this script unless the SendMessage function above changed something
200  if (mFactor == 1f && mAmountPerDelta > 0f || mFactor == 0f && mAmountPerDelta < 0f)
201  {
202  enabled = false;
203  }
204  }
205  }
206  }
