50  void RepositionVariableSize (List<Transform> children)
51  {
52  float xOffset = 0;
53  float yOffset = 0;
54 
55  int cols = columns > 0 ? children.Count / columns + 1 : 1;
56  int rows = columns > 0 ? columns : children.Count;
57 
58  Bounds[,] bounds = new Bounds[cols, rows];
59  Bounds[] boundsRows = new Bounds[rows];
60  Bounds[] boundsCols = new Bounds[cols];
61 
62  int x = 0;
63  int y = 0;
64 
65  for (int i = 0, imax = children.Count; i < imax; ++i)
66  {
67  Transform t = children[i];
68  Bounds b = NGUIMath.CalculateRelativeWidgetBounds(t);
69  Vector3 scale = t.localScale;
70  b.min = Vector3.Scale(b.min, scale);
71  b.max = Vector3.Scale(b.max, scale);
72  bounds[y, x] = b;
73 
74  boundsRows[x].Encapsulate(b);
75  boundsCols[y].Encapsulate(b);
76 
77  if (++x >= columns && columns > 0)
78  {
79  x = 0;
80  ++y;
81  }
82  }
83 
84  x = 0;
85  y = 0;
86 
87  for (int i = 0, imax = children.Count; i < imax; ++i)
88  {
89  Transform t = children[i];
90  Bounds b = bounds[y, x];
91  Bounds br = boundsRows[x];
92  Bounds bc = boundsCols[y];
93 
94  Vector3 pos = t.localPosition;
95  pos.x = xOffset + b.extents.x - b.center.x;
96  pos.x += b.min.x - br.min.x + padding.x;
97 
98  if (direction == Direction.Down)
99  {
100  pos.y = -yOffset - b.extents.y - b.center.y;
101  pos.y += (b.max.y - b.min.y - bc.max.y + bc.min.y) * 0.5f - padding.y;
102  }
103  else
104  {
105  pos.y = yOffset + b.extents.y - b.center.y;
106  pos.y += (b.max.y - b.min.y - bc.max.y + bc.min.y) * 0.5f - padding.y;
107  }
108 
109  xOffset += br.max.x - br.min.x + padding.x * 2f;
110 
111  t.localPosition = pos;
112 
113  if (++x >= columns && columns > 0)
114  {
115  x = 0;
116  ++y;
117 
118  xOffset = 0f;
119  yOffset += bc.size.y + padding.y * 2f;
120  }
121  }
122  }

128  public void Reposition ()
129  {
130  if (mStarted)
131  {
132  Transform myTrans = transform;
133  List<Transform> children = new List<Transform>();
134 
135  for (int i = 0; i < myTrans.childCount; ++i)
136  {
137  Transform child = myTrans.GetChild(i);
138  if (child && (!hideInactive || child.gameObject.active)) children.Add(child);
139  }
140  if (sorted) children.Sort(SortByName);
141  if (children.Count > 0) RepositionVariableSize(children);
142  if (mPanel != null && mDrag == null) mPanel.ConstrainTargetToBounds(myTrans, true);
143  if (mDrag != null) mDrag.UpdateScrollbars(true);
144  }
145  else repositionNow = true;
146  }
