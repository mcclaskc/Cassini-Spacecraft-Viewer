96  bool AdjustRadial (Vector2[] xy, Vector2[] uv, float fill, bool invert)
97  {
98  // Nothing to fill
99  if (fill < 0.001f) return false;
100 
101  // Nothing to adjust
102  if (!invert && fill > 0.999f) return true;
103 
104  // Convert 0-1 value into 0 to 90 degrees angle in radians
105  float angle = Mathf.Clamp01(fill);
106  if (!invert) angle = 1f - angle;
107  angle *= 90f * Mathf.Deg2Rad;
108 
109  // Calculate the effective X and Y factors
110  float fx = Mathf.Sin(angle);
111  float fy = Mathf.Cos(angle);
112 
113  // Normalize the result, so it's projected onto the side of the rectangle
114  if (fx > fy)
115  {
116  fy *= 1f / fx;
117  fx = 1f;
118 
119  if (!invert)
120  {
121  xy[0].y = Mathf.Lerp(xy[2].y, xy[0].y, fy);
122  xy[3].y = xy[0].y;
123 
124  uv[0].y = Mathf.Lerp(uv[2].y, uv[0].y, fy);
125  uv[3].y = uv[0].y;
126  }
127  }
128  else if (fy > fx)
129  {
130  fx *= 1f / fy;
131  fy = 1f;
132 
133  if (invert)
134  {
135  xy[0].x = Mathf.Lerp(xy[2].x, xy[0].x, fx);
136  xy[1].x = xy[0].x;
137 
138  uv[0].x = Mathf.Lerp(uv[2].x, uv[0].x, fx);
139  uv[1].x = uv[0].x;
140  }
141  }
142  else
143  {
144  fx = 1f;
145  fy = 1f;
146  }
147 
148  if (invert)
149  {
150  xy[1].y = Mathf.Lerp(xy[2].y, xy[0].y, fy);
151  uv[1].y = Mathf.Lerp(uv[2].y, uv[0].y, fy);
152  }
153  else
154  {
155  xy[3].x = Mathf.Lerp(xy[2].x, xy[0].x, fx);
156  uv[3].x = Mathf.Lerp(uv[2].x, uv[0].x, fx);
157  }
158  return true;
159  }

166  {
167  for (int i = 0; i < offset; ++i)
168  {
169  Vector2 v0 = new Vector2(v[3].x, v[3].y);
170 
171  v[3].x = v[2].y;
172  v[3].y = v[2].x;
173 
174  v[2].x = v[1].y;
175  v[2].y = v[1].x;
176 
177  v[1].x = v[0].y;
178  v[1].y = v[0].x;
179 
180  v[0].x = v0.y;
181  v[0].y = v0.x;
182  }

189  override public void OnFill (BetterList<Vector3> verts, BetterList<Vector2> uvs, BetterList<Color> cols)
190  {
191  float x0 = 0f;
192  float y0 = 0f;
193  float x1 = 1f;
194  float y1 = -1f;
195 
196  float u0 = mOuterUV.xMin;
197  float v0 = mOuterUV.yMin;
198  float u1 = mOuterUV.xMax;
199  float v1 = mOuterUV.yMax;
200 
201  // Horizontal and vertical filled sprites are simple -- just end the sprite prematurely
202  if (mFillDirection == FillDirection.Horizontal || mFillDirection == FillDirection.Vertical)
203  {
204  float du = (u1 - u0) * mFillAmount;
205  float dv = (v1 - v0) * mFillAmount;
206 
207  if (fillDirection == FillDirection.Horizontal)
208  {
209  if (mInvert)
210  {
211  x0 = (1f - mFillAmount);
212  u0 = u1 - du;
213  }
214  else
215  {
216  x1 *= mFillAmount;
217  u1 = u0 + du;
218  }
219  }
220  else if (fillDirection == FillDirection.Vertical)
221  {
222  if (mInvert)
223  {
224  y1 *= mFillAmount;
225  v0 = v1 - dv;
226  }
227  else
228  {
229  y0 = -(1f - mFillAmount);
230  v1 = v0 + dv;
231  }
232  }
233  }
234 
235  // Starting quad for the sprite
236  Vector2[] xy = new Vector2[4];
237  Vector2[] uv = new Vector2[4];
238 
239  xy[0] = new Vector2(x1, y0);
240  xy[1] = new Vector2(x1, y1);
241  xy[2] = new Vector2(x0, y1);
242  xy[3] = new Vector2(x0, y0);
243 
244  uv[0] = new Vector2(u1, v1);
245  uv[1] = new Vector2(u1, v0);
246  uv[2] = new Vector2(u0, v0);
247  uv[3] = new Vector2(u0, v1);
248 
249  if (fillDirection == FillDirection.Radial90)
250  {
251  // Adjust the quad radially, and if 'false' is returned (it's not visible), just exit
252  if (!AdjustRadial(xy, uv, mFillAmount, mInvert)) return;
253  }
254  else if (fillDirection == FillDirection.Radial180)
255  {
256  // Working in 0-1 coordinates is easier
257  Vector2[] oxy = new Vector2[4];
258  Vector2[] ouv = new Vector2[4];
259 
260  for (int i = 0; i < 2; ++i)
261  {
262  oxy[0] = new Vector2(0f, 0f);
263  oxy[1] = new Vector2(0f, 1f);
264  oxy[2] = new Vector2(1f, 1f);
265  oxy[3] = new Vector2(1f, 0f);
266 
267  ouv[0] = new Vector2(0f, 0f);
268  ouv[1] = new Vector2(0f, 1f);
269  ouv[2] = new Vector2(1f, 1f);
270  ouv[3] = new Vector2(1f, 0f);
271 
272  // Each half must be rotated 90 degrees clockwise in order for it to fill properly
273  if (mInvert)
274  {
275  if (i > 0)
276  {
277  Rotate(oxy, i);
278  Rotate(ouv, i);
279  }
280  }
281  else if (i < 1)
282  {
283  Rotate(oxy, 1 - i);
284  Rotate(ouv, 1 - i);
285  }
286 
287  // Each half must fill in only a part of the space
288  float x, y;
289 
290  if (i == 1)
291  {
292  x = mInvert ? 0.5f : 1f;
293  y = mInvert ? 1f : 0.5f;
294  }
295  else
296  {
297  x = mInvert ? 1f : 0.5f;
298  y = mInvert ? 0.5f : 1f;
299  }
300 
301  oxy[1].y = Mathf.Lerp(x, y, oxy[1].y);
302  oxy[2].y = Mathf.Lerp(x, y, oxy[2].y);
303  ouv[1].y = Mathf.Lerp(x, y, ouv[1].y);
304  ouv[2].y = Mathf.Lerp(x, y, ouv[2].y);
305 
306  float amount = (mFillAmount) * 2 - i;
307  bool odd = (i % 2) == 1;
308 
309  if (AdjustRadial(oxy, ouv, amount, !odd))
310  {
311  if (mInvert) odd = !odd;
312 
313  // Add every other side in reverse order so they don't come out backface-culled due to rotation
314  if (odd)
315  {
316  for (int b = 0; b < 4; ++b)
317  {
318  x = Mathf.Lerp(xy[0].x, xy[2].x, oxy[b].x);
319  y = Mathf.Lerp(xy[0].y, xy[2].y, oxy[b].y);
320 
321  float u = Mathf.Lerp(uv[0].x, uv[2].x, ouv[b].x);
322  float v = Mathf.Lerp(uv[0].y, uv[2].y, ouv[b].y);
323 
324  verts.Add(new Vector3(x, y, 0f));
325  uvs.Add(new Vector2(u, v));
326  cols.Add(color);
327  }
328  }
329  else
330  {
331  for (int b = 3; b > -1; --b)
332  {
333  x = Mathf.Lerp(xy[0].x, xy[2].x, oxy[b].x);
334  y = Mathf.Lerp(xy[0].y, xy[2].y, oxy[b].y);
335 
336  float u = Mathf.Lerp(uv[0].x, uv[2].x, ouv[b].x);
337  float v = Mathf.Lerp(uv[0].y, uv[2].y, ouv[b].y);
338 
339  verts.Add(new Vector3(x, y, 0f));
340  uvs.Add(new Vector2(u, v));
341  cols.Add(color);
342  }
343  }
344  }
345  }
346  return;
347  }
348  else if (fillDirection == FillDirection.Radial360)
349  {
350  float[] matrix = new float[]
351  {
352  // x0 y0 x1 y1
353  0.5f, 1f, 0f, 0.5f, // quadrant 0
354  0.5f, 1f, 0.5f, 1f, // quadrant 1
355  0f, 0.5f, 0.5f, 1f, // quadrant 2
356  0f, 0.5f, 0f, 0.5f, // quadrant 3
357  };
358 
359  Vector2[] oxy = new Vector2[4];
360  Vector2[] ouv = new Vector2[4];
361 
362  for (int i = 0; i < 4; ++i)
363  {
364  oxy[0] = new Vector2(0f, 0f);
365  oxy[1] = new Vector2(0f, 1f);
366  oxy[2] = new Vector2(1f, 1f);
367  oxy[3] = new Vector2(1f, 0f);
368 
369  ouv[0] = new Vector2(0f, 0f);
370  ouv[1] = new Vector2(0f, 1f);
371  ouv[2] = new Vector2(1f, 1f);
372  ouv[3] = new Vector2(1f, 0f);
373 
374  // Each quadrant must be rotated 90 degrees clockwise in order for it to fill properly
375  if (mInvert)
376  {
377  if (i > 0)
378  {
379  Rotate(oxy, i);
380  Rotate(ouv, i);
381  }
382  }
383  else if (i < 3)
384  {
385  Rotate(oxy, 3 - i);
386  Rotate(ouv, 3 - i);
387  }
388 
389  // Each quadrant must fill in only a quarter of the space
390  for (int b = 0; b < 4; ++b)
391  {
392  int index = (mInvert) ? (3 - i) * 4 : i * 4;
393 
394  float fx0 = matrix[index];
395  float fy0 = matrix[index + 1];
396  float fx1 = matrix[index + 2];
397  float fy1 = matrix[index + 3];
398 
399  oxy[b].x = Mathf.Lerp(fx0, fy0, oxy[b].x);
400  oxy[b].y = Mathf.Lerp(fx1, fy1, oxy[b].y);
401  ouv[b].x = Mathf.Lerp(fx0, fy0, ouv[b].x);
402  ouv[b].y = Mathf.Lerp(fx1, fy1, ouv[b].y);
403  }
404 
405  float amount = (mFillAmount) * 4 - i;
406  bool odd = (i % 2) == 1;
407 
408  if (AdjustRadial(oxy, ouv, amount, !odd))
409  {
410  if (mInvert) odd = !odd;
411 
412  // Add every other side in reverse order so they don't come out backface-culled due to rotation
413  if (odd)
414  {
415  for (int b = 0; b < 4; ++b)
416  {
417  float x = Mathf.Lerp(xy[0].x, xy[2].x, oxy[b].x);
418  float y = Mathf.Lerp(xy[0].y, xy[2].y, oxy[b].y);
419  float u = Mathf.Lerp(uv[0].x, uv[2].x, ouv[b].x);
420  float v = Mathf.Lerp(uv[0].y, uv[2].y, ouv[b].y);
421
