164  public void Drag (Vector2 delta)
165  {
166  UICamera.currentTouch.clickNotification = UICamera.ClickNotification.BasedOnDelta;
167 
168  if (mRoot != null && !mRoot.automatic) delta *= (float)mRoot.manualHeight / Screen.height;
169 
170  Vector2 offset = Vector2.Scale(delta, -scale);
171  mTrans.localPosition += (Vector3)offset;
172 
173  // Adjust the momentum
174  mMomentum = Vector2.Lerp(mMomentum, mMomentum + offset * (0.01f * momentumAmount), 0.67f);
175 
176  // Constrain the UI to the bounds, and if done so, eliminate the momentum
177  if (dragEffect != UIDragObject.DragEffect.MomentumAndSpring && ConstrainToBounds(true))
178  {
179  mMomentum = Vector2.zero;
180  mScroll = 0f;
181  }
182  }

201  void Update ()
202  {
203  float delta = UpdateRealTimeDelta();
204 
205  if (mPressed)
206  {
207  // Disable the spring movement
208  SpringPosition sp = GetComponent<SpringPosition>();
209  if (sp != null) sp.enabled = false;
210  mScroll = 0f;
211  }
212  else
213  {
214  mMomentum += scale * (mScroll * 20f);
215  mScroll = NGUIMath.SpringLerp(mScroll, 0f, 20f, delta);
216 
217  if (mMomentum.magnitude > 0.01f)
218  {
219  // Apply the momentum
220  mTrans.localPosition += (Vector3)NGUIMath.SpringDampen(ref mMomentum, 9f, delta);
221  mBounds = NGUIMath.CalculateAbsoluteWidgetBounds(rootForBounds);
222 
223  if (!ConstrainToBounds(dragEffect == UIDragObject.DragEffect.None))
224  {
225  SpringPosition sp = GetComponent<SpringPosition>();
226  if (sp != null) sp.enabled = false;
227  }
228  return;
229  }
230  else mScroll = 0f;
231  }
232
