112  mSize.x *= textScale.x;

114  mSize.x += background.border.x + background.border.z + ( offset.x - background.border.x) * 2f;
115  mSize.y += background.border.y + background.border.w + (-offset.y - background.border.y) * 2f;

121  if (uiCamera != null)
122  {
123  // Since the screen can be of different than expected size, we want to convert
124  // mouse coordinates to view space, then convert that to world position.
125  mPos.x = Mathf.Clamp01(mPos.x / Screen.width);
126  mPos.y = Mathf.Clamp01(mPos.y / Screen.height);
127 
128  // Calculate the ratio of the camera's target orthographic size to current screen size
129  float activeSize = uiCamera.orthographicSize / mTrans.parent.lossyScale.y;
130  float ratio = (Screen.height * 0.5f) / activeSize;
131 
132  // Calculate the maximum on-screen size of the tooltip window
133  Vector2 max = new Vector2(ratio * mSize.x / Screen.width, ratio * mSize.y / Screen.height);
134 
135  // Limit the tooltip to always be visible
136  mPos.x = Mathf.Min(mPos.x, 1f - max.x);
137  mPos.y = Mathf.Max(mPos.y, max.y);

141  mPos = mTrans.localPosition;

145  }

150  if (mPos.y - mSize.y < 0f) mPos.y = mSize.y;

152  // Simple calculation that assumes that the camera is of fixed size

155  }
