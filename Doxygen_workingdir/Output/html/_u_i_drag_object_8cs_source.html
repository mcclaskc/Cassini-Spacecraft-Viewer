81  void OnPress (bool pressed)
82  {
83  if (enabled && gameObject.active && target != null)
84  {
85  mPressed = pressed;
86 
87  if (pressed)
88  {
89  if (restrictWithinPanel && mPanel == null) FindPanel();
90 
91  // Calculate the bounds
92  if (restrictWithinPanel) mBounds = NGUIMath.CalculateRelativeWidgetBounds(mPanel.cachedTransform, target);
93 
94  // Remove all momentum on press
95  mMomentum = Vector3.zero;
96  mScroll = 0f;
97 
98  // Disable the spring movement
99  SpringPosition sp = target.GetComponent<SpringPosition>();
100  if (sp != null) sp.enabled = false;
101 
102  // Remember the hit position
103  mLastPos = UICamera.lastHit.point;
104 
105  // Create the plane to drag along
106  Transform trans = UICamera.currentCamera.transform;
107  mPlane = new Plane((mPanel != null ? mPanel.cachedTransform.rotation : trans.rotation) * Vector3.back, mLastPos);
108  }
109  else if (restrictWithinPanel && mPanel.clipping != UIDrawCall.Clipping.None && dragEffect == DragEffect.MomentumAndSpring)
110  {
111  mPanel.ConstrainTargetToBounds(target, ref mBounds, false);
112  }
113  }

120  void OnDrag (Vector2 delta)
121  {
122  if (enabled && gameObject.active && target != null)
123  {
124  UICamera.currentTouch.clickNotification = UICamera.ClickNotification.BasedOnDelta;
125 
126  Ray ray = UICamera.currentCamera.ScreenPointToRay(UICamera.currentTouch.pos);
127  float dist = 0f;
128 
129  if (mPlane.Raycast(ray, out dist))
130  {
131  Vector3 currentPos = ray.GetPoint(dist);
132  Vector3 offset = currentPos - mLastPos;
133  mLastPos = currentPos;
134 
135  if (offset.x != 0f || offset.y != 0f)
136  {
137  offset = target.InverseTransformDirection(offset);
138  offset.Scale(scale);
139  offset = target.TransformDirection(offset);
140  }
141 
142  // Adjust the momentum
143  mMomentum = Vector3.Lerp(mMomentum, mMomentum + offset * (0.01f * momentumAmount), 0.67f);
144 
145  // We want to constrain the UI to be within bounds
146  if (restrictWithinPanel)
147  {
148  // Adjust the position and bounds
149  Vector3 localPos = target.localPosition;
150  target.position += offset;
151  mBounds.center = mBounds.center + (target.localPosition - localPos);
152 
153  // Constrain the UI to the bounds, and if done so, eliminate the momentum
154  if (dragEffect != DragEffect.MomentumAndSpring && mPanel.clipping != UIDrawCall.Clipping.None &&
155  mPanel.ConstrainTargetToBounds(target, ref mBounds, true))
156  {
157  mMomentum = Vector3.zero;
158  mScroll = 0f;
159  }
160  }
161  else
162  {
163  // Adjust the position
164  target.position += offset;
165  }
166  }
167  }
168  }

174  void LateUpdate ()
175  {
176  float delta = UpdateRealTimeDelta();
177  if (target == null) return;
178 
179  if (mPressed)
180  {
181  // Disable the spring movement
182  SpringPosition sp = target.GetComponent<SpringPosition>();
183  if (sp != null) sp.enabled = false;
184  mScroll = 0f;
185  }
186  else
187  {
188  mMomentum += scale * (-mScroll * 0.05f);
189  mScroll = NGUIMath.SpringLerp(mScroll, 0f, 20f, delta);
190 
191  if (mMomentum.magnitude > 0.0001f)
192  {
193  // Apply the momentum
194  if (mPanel == null) FindPanel();
195 
196  if (mPanel != null)
197  {
198  target.position += NGUIMath.SpringDampen(ref mMomentum, 9f, delta);
199 
200  if (restrictWithinPanel && mPanel.clipping != UIDrawCall.Clipping.None)
201  {
202  mBounds = NGUIMath.CalculateRelativeWidgetBounds(mPanel.cachedTransform, target);
203 
204  if (!mPanel.ConstrainTargetToBounds(target, ref mBounds, dragEffect == DragEffect.None))
205  {
206  SpringPosition sp = target.GetComponent<SpringPosition>();
207  if (sp != null) sp.enabled = false;
208  }
209  }
210  return;
211  }
212  }
213  else mScroll = 0f;
214  }
215 
216  // Dampen the momentum
217  NGUIMath.SpringDampen(ref mMomentum, 9f, delta);
218  }
