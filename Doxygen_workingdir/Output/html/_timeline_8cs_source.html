226  //Make sure this level should be visible
227  if(visibleRange.TotalDays < dayStartRange) {
228 
229  //Figure out where the height of the text should be
230  float ratio = (dayStartRange - (float)visibleRange.TotalDays) / (dayStartRange - dayEndRange);
231  ratio = ratio > 1.0f ? 1.0f : ratio;
232  float yOffset = timelineY + finalY + (1-ratio) * (timelineHeight - finalY + 10.0f);
