132  void Update ()
133  {
134  if (disableWhenFinished != DisableCondition.DoNotDisable && mTweens != null)
135  {

137  bool properDirection = true;
138 
139  for (int i = 0, imax = mTweens.Length; i < imax; ++i)
140  {
141  UITweener tw = mTweens[i];
142 
143  if (tw.enabled)
144  {
145  isFinished = false;
146  break;
147  }

149  {
150  properDirection = false;

166  public void Play (bool forward)
167  {
168  GameObject go = (tweenTarget == null) ? gameObject : tweenTarget;
169 
170  if (!go.active)
171  {
172  // If the object is disabled, don't do anything
173  if (ifDisabledOnPlay != EnableCondition.EnableThenPlay) return;
174 
175  // Enable the game object before tweening it
176  NGUITools.SetActive(go, true);
177  }
178 
179  // Gather the tweening components
180  mTweens = includeChildren ? go.GetComponentsInChildren<UITweener>() : go.GetComponents<UITweener>();
181 
182  if (mTweens.Length == 0)
183  {
184  // No tweeners found -- should we disable the object?
185  if (disableWhenFinished != DisableCondition.DoNotDisable) NGUITools.SetActive(tweenTarget, false);
186  }
187  else
188  {
189  bool activated = false;
190  if (playDirection == Direction.Reverse) forward = !forward;
191 
192  // Run through all located tween components
193  for (int i = 0, imax = mTweens.Length; i < imax; ++i)
194  {
195  UITweener tw = mTweens[i];
196 
197  // If the tweener's group matches, we can work with it
198  if (tw.tweenGroup == tweenGroup)
199  {
200  // Ensure that the game objects are enabled
201  if (!activated && !go.active)
202  {
203  activated = true;
204  NGUITools.SetActive(go, true);
205  }
206 
207  // Toggle or activate the tween component
208  if (playDirection == Direction.Toggle) tw.Toggle();
209  else tw.Play(forward);
210  if (resetOnPlay) tw.Reset();
211 
212  // Copy the event receiver
213  if (eventReceiver != null && !string.IsNullOrEmpty(callWhenFinished))
214  {
215  tw.eventReceiver = eventReceiver;
216  tw.callWhenFinished = callWhenFinished;
217  }
218  }
219  }
