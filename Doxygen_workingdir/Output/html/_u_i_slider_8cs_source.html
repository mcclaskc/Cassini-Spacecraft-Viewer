167  void OnKey (KeyCode key)
168  {
169  float step = (numberOfSteps > 1f) ? 1f / (numberOfSteps - 1) : 0.125f;
170 
171  if (direction == Direction.Horizontal)
172  {
173  if (key == KeyCode.LeftArrow) Set(rawValue - step, false);
174  else if (key == KeyCode.RightArrow) Set(rawValue + step, false);
175  }
176  else
177  {
178  if (key == KeyCode.DownArrow) Set(rawValue - step, false);
179  else if (key == KeyCode.UpArrow) Set(rawValue + step, false);

187  void UpdateDrag ()
188  {
189  // Create a plane for the slider
190  if (mCol == null || UICamera.currentCamera == null || UICamera.currentTouch == null) return;
191 
192  // Don't consider the slider for click events
193  UICamera.currentTouch.clickNotification = UICamera.ClickNotification.None;
194 
195  // Create a ray and a plane
196  Ray ray = UICamera.currentCamera.ScreenPointToRay(UICamera.currentTouch.pos);
197  Plane plane = new Plane(mTrans.rotation * Vector3.back, mTrans.position);
198 
199  // If the ray doesn't hit the plane, do nothing
200  float dist;
201  if (!plane.Raycast(ray, out dist)) return;
202 
203  // Collider's bottom-left corner in local space
204  Vector3 localOrigin = mTrans.localPosition + mCol.center - mCol.extents;
205  Vector3 localOffset = mTrans.localPosition - localOrigin;
206 
207  // Direction to the point on the plane in scaled local space
208  Vector3 localCursor = mTrans.InverseTransformPoint(ray.GetPoint(dist));
209  Vector3 dir = localCursor + localOffset;
210 
211  // Update the slider
212  Set( (direction == Direction.Horizontal) ? dir.x / mCol.size.x : dir.y / mCol.size.y, false );
213  }

219  void Set (float input, bool force)
220  {
221  if (!mInitDone) Init();
222 
223  // Clamp the input
224  float val = Mathf.Clamp01(input);
225  if (val < 0.001f) val = 0f;
226 
227  // Save the raw value
228  rawValue = val;
229 
230  // Take steps into consideration
231  if (numberOfSteps > 1) val = Mathf.Round(val * (numberOfSteps - 1)) / (numberOfSteps - 1);
232 
233  // If the stepped value doesn't match the last one, it's time to update
234  if (force || mStepValue != val)
235  {
236  mStepValue = val;
237  Vector3 scale = fullSize;
238 
239  if (direction == Direction.Horizontal) scale.x *= mStepValue;
240  else scale.y *= mStepValue;
241 
242  if (mFGFilled != null)
243  {
244  mFGFilled.fillAmount = mStepValue;
245  }
246  else if (foreground != null)
247  {
248  mFGTrans.localScale = scale;
249 
250  if (mFGWidget != null)
251  {
252  if (val > 0.001f)
253  {
254  mFGWidget.enabled = true;
255  mFGWidget.MarkAsChanged();
256  }
257  else
258  {
259  mFGWidget.enabled = false;
260  }
261  }
262  }
263 
264  if (thumb != null)
265  {
266  Vector3 pos = thumb.localPosition;
267 
268  if (mFGFilled != null)
269  {
270  if (mFGFilled.fillDirection == UIFilledSprite.FillDirection.Horizontal)
271  {
272  pos.x = mFGFilled.invert ? fullSize.x - scale.x : scale.x;
273  }
274  else if (mFGFilled.fillDirection == UIFilledSprite.FillDirection.Vertical)
275  {
276  pos.y = mFGFilled.invert ? fullSize.y - scale.y : scale.y;
277  }
278  }
279  else if (direction == Direction.Horizontal)
280  {
281  pos.x = scale.x;
282  }
283  else
284  {
285  pos.y = scale.y;
286  }
287  thumb.localPosition = pos;
288  }
289 
290  if (eventReceiver != null && !string.IsNullOrEmpty(functionName) && Application.isPlaying)
291  {
292  current = this;
293  eventReceiver.SendMessage(functionName, mStepValue, SendMessageOptions.DontRequireReceiver);
294  current = null;
295  }
296  }
