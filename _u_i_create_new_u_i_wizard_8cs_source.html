64  void CreateNewUI ()
65  {
66  NGUIEditorTools.RegisterUndo("Create New UI");
67 
68  // Root for the UI
69  GameObject root = null;
70 
71  if (camType == CameraType.Simple2D)
72  {
73  root = new GameObject("UI Root (2D)");
74  root.AddComponent<UIRoot>();
75  }
76  else
77  {
78  root = new GameObject((camType == CameraType.Advanced3D) ? "UI Root (3D)" : "UI Root");
79  root.transform.localScale = new Vector3(0.0025f, 0.0025f, 0.0025f);
80 
81  UIRoot uiRoot = root.AddComponent<UIRoot>();
82  uiRoot.automatic = false;
83  uiRoot.manualHeight = 800;
84  }
85 
86  // Assign the layer to be used by everything
87  root.layer = layer;
88 
89  // Figure out the depth of the highest camera
90  if (camType == CameraType.None)
91  {
92  // No camera requested -- simply add a panel
93  UIPanel panel = NGUITools.AddChild<UIPanel>(root.gameObject);
94  Selection.activeGameObject = panel.gameObject;
95  }
96  else
97  {
98  int mask = 1 << layer;
99  float depth = -1f;
100  bool clearColor = true;
101  bool audioListener = true;
102 
103  List<Camera> cameras = NGUIEditorTools.FindInScene<Camera>();
104 
105  foreach (Camera c in cameras)
106  {
107  // Choose the maximum depth
108  depth = Mathf.Max(depth, c.depth);
109 
110  // Automatically exclude the specified layer mask from the camera if it can see more than that layer
111  if (layer != 0 && c.cullingMask != mask) c.cullingMask = (c.cullingMask & (~mask));
112 
113  // Only consider this object if it's active
114  if (c.enabled && c.gameObject.active) clearColor = false;
115 
116  // If this camera has an audio listener, we won't need to add one
117  if (c.GetComponent<AudioListener>() != null) audioListener = false;
118  }
119 
120  // Camera and UICamera for this UI
121  Camera cam = NGUITools.AddChild<Camera>(root);
122  cam.depth = depth + 1;
123  cam.backgroundColor = Color.grey;
124  cam.cullingMask = mask;
125 
126  if (camType == CameraType.Simple2D)
127  {
128  cam.orthographicSize = 1f;
129  cam.orthographic = true;
130  cam.nearClipPlane = -2f;
131  cam.farClipPlane = 2f;
132  }
133  else
134  {
135  cam.nearClipPlane = 0.1f;
136  cam.farClipPlane = 4f;
137  cam.transform.localPosition = new Vector3(0f, 0f, -1.7f);
138  }
139 
140  // We don't want to clear color if this is not the first camera
141  if (cameras.Count > 0) cam.clearFlags = clearColor ? CameraClearFlags.Skybox : CameraClearFlags.Depth;
142 
143  // Add an audio listener if we need one
144  if (audioListener) cam.gameObject.AddComponent<AudioListener>();
145 
146  // Add a UI Camera for event handling
147  cam.gameObject.AddComponent<UICamera>();
148 
149  // Anchor is useful to have
150  UIAnchor anchor = NGUITools.AddChild<UIAnchor>(cam.gameObject);
151  anchor.uiCamera = cam;
152 
153  // Since the camera was brought back 700 units above, we should bring the anchor forward to compensate
154  if (camType == CameraType.Advanced3D) anchor.depthOffset = 1.7f;
155 
156  // And finally -- the first UI panel
157  UIPanel panel = NGUITools.AddChild<UIPanel>(anchor.gameObject);
158  Selection.activeGameObject = panel.gameObject;
159  }
160  }
