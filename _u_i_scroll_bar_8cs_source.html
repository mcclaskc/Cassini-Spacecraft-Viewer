191  void CenterOnPos (Vector2 localPos)
192  {
193  if (mBG == null || mFG == null) return;
194 
195  // Background's bounds
196  Bounds bg = NGUIMath.CalculateRelativeInnerBounds(cachedTransform, mBG);
197  Bounds fg = NGUIMath.CalculateRelativeInnerBounds(cachedTransform, mFG);
198 
199  if (mDir == Direction.Horizontal)
200  {
201  float size = bg.size.x - fg.size.x;
202  float offset = size * 0.5f;
203  float min = bg.center.x - offset;
204  float val = (size > 0f) ? (localPos.x - min) / size : 0f;
205  scrollValue = mInverted ? 1f - val : val;
206  }
207  else
208  {
209  float size = bg.size.y - fg.size.y;
210  float offset = size * 0.5f;
211  float min = bg.center.y - offset;
212  float val = (size > 0f) ? 1f - (localPos.y - min) / size : 0f;
213  scrollValue = mInverted ? 1f - val : val;
214  }
215  }

221  void Reposition (Vector2 screenPos)
222  {
223  // Create a plane
224  Transform trans = cachedTransform;
225  Plane plane = new Plane(trans.rotation * Vector3.back, trans.position);
226 
227  // If the ray doesn't hit the plane, do nothing
228  float dist;
229  Ray ray = cachedCamera.ScreenPointToRay(screenPos);
230  if (!plane.Raycast(ray, out dist)) return;
231 
232  // Transform the point from world space to local space
233  CenterOnPos(trans.InverseTransformPoint(ray.GetPoint(dist)));
234  }

312  public void ForceUpdate ()
313  {
314  mIsDirty = false;
315 
316  if (mBG != null && mFG != null)
317  {
318  mSize = Mathf.Clamp01(mSize);
319  mScroll = Mathf.Clamp01(mScroll);
320 
321  Vector4 bg = mBG.border;
322  Vector4 fg = mFG.border;
323 
324  // Space available for the background
325  Vector2 bgs = new Vector2(
326  Mathf.Max(0f, mBG.cachedTransform.localScale.x - bg.x - bg.z),
327  Mathf.Max(0f, mBG.cachedTransform.localScale.y - bg.y - bg.w));
328 
329  float val = mInverted ? 1f - mScroll : mScroll;
330 
331  if (mDir == Direction.Horizontal)
332  {
333  Vector2 fgs = new Vector2(bgs.x * mSize, bgs.y);
334 
335  mFG.pivot = UIWidget.Pivot.Left;
336  mBG.pivot = UIWidget.Pivot.Left;
337  mBG.cachedTransform.localPosition = Vector3.zero;
338  mFG.cachedTransform.localPosition = new Vector3(bg.x - fg.x + (bgs.x - fgs.x) * val, 0f, 0f);
339  mFG.cachedTransform.localScale = new Vector3(fgs.x + fg.x + fg.z, fgs.y + fg.y + fg.w, 1f);
340  if (val < 0.999f && val > 0.001f) mFG.MakePixelPerfect();
341  }
342  else
343  {
344  Vector2 fgs = new Vector2(bgs.x, bgs.y * mSize);
345 
346  mFG.pivot = UIWidget.Pivot.Top;
347  mBG.pivot = UIWidget.Pivot.Top;
348  mBG.cachedTransform.localPosition = Vector3.zero;
349  mFG.cachedTransform.localPosition = new Vector3(0f, -bg.y + fg.y - (bgs.y - fgs.y) * val, 0f);
350  mFG.cachedTransform.localScale = new Vector3(fgs.x + fg.x + fg.z, fgs.y + fg.y + fg.w, 1f);
351  if (val < 0.999f && val > 0.001f) mFG.MakePixelPerfect();
352  }
353  }
354  }
