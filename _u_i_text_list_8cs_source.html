124  protected void UpdateVisibleText ()
125  {
126  if (textLabel != null)
127  {
128  UIFont font = (UIFont)textLabel.font;
129 
130  if (font != null)
131  {
132  int lines = 0;
133  int maxLines = maxHeight > 0 ? Mathf.FloorToInt(maxHeight / textLabel.cachedTransform.localScale.y) : 100000;
134  int offset = Mathf.RoundToInt(mScroll);
135 
136  // Don't let scrolling to exceed the visible number of lines
137  if (maxLines + offset > mTotalLines)
138  {
139  offset = Mathf.Max(0, mTotalLines - maxLines);
140  mScroll = offset;
141  }
142 
143  if (style == Style.Chat)
144  {
145  offset = Mathf.Max(0, mTotalLines - maxLines - offset);
146  }
147 
148  string final = "";
149 
150  for (int i = 0, imax = mParagraphs.Count; i < imax; ++i)
151  {
152  Paragraph p = mParagraphs[i];
153 
154  for (int b = 0, bmax = p.lines.Length; b < bmax; ++b)
155  {
156  string s = p.lines[b];
157 
158  if (offset > 0)
159  {
160  --offset;
161  }
162  else
163  {
164  if (final.Length > 0) final += "\n";
165  final += s;
166  ++lines;
167  if (lines >= maxLines) break;
168  }
169  }
170  if (lines >= maxLines) break;
171  }
172  textLabel.text = final;
173  }
