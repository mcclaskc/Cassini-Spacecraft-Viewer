14  void ValidatePivot (Transform fg, string name, UISlider.Direction dir)
15  {
16  if (fg != null)
17  {
18  UIWidget widget = fg.GetComponent<UIWidget>();
19 
20  if (widget != null && !(widget is UIFilledSprite))
21  {
22  if (dir == UISlider.Direction.Horizontal)
23  {
24  if (widget.pivot != UIWidget.Pivot.Left &&
25  widget.pivot != UIWidget.Pivot.TopLeft &&
26  widget.pivot != UIWidget.Pivot.BottomLeft)
27  {
28  GUI.color = new Color(1f, 0.7f, 0f);
29  GUILayout.Label(name + " should use a Left pivot");
30  GUI.color = Color.white;
31  }
32  }
33  else if (widget.pivot != UIWidget.Pivot.BottomLeft &&
34  widget.pivot != UIWidget.Pivot.Bottom &&
35  widget.pivot != UIWidget.Pivot.BottomRight)
36  {
37  GUI.color = new Color(1f, 0.7f, 0f);
38  GUILayout.Label(name + " should use a Bottom pivot");
39  GUI.color = Color.white;
40  }
41  }
42  }
